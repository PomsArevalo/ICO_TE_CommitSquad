    align-items: center;
    justify-content: space-between;
}
/doy estilo a los botones (tipo de letra, tamaño, y le agrego una transición)/
.botones a{
    display: inline-block;
    position: relative;
    margin-right: 50px; 
    text-decoration: none;
    font-size: 18px;
    font-family: "Nunito", sans-serif;
    color: black;
    transition: background-color 0.3s, color 0.3s;
}
/agrego estilo a mis botones (el color que tendrán las letras cuando pase el mouse)/
.botones a:hover {
    background-color: none; 
    color: orange;            
}
/agrego estilo a mis botones (creo la línea debajo del enlace cuando el mouse está sobre el)/
.botones a::after {
    content: "";  
    position: absolute;
    left: 0;
    bottom: -2px;  
    width: 0;
    height: 3px;  
    background-color: orange;
    transition: width 0.3s;
}
/Le doy estilo a la línea debajo de los botones/
.botones a:hover::after {
    width: 100%;  
}
